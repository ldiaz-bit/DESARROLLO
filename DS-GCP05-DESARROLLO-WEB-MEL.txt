            <script>
                function calcular(){
                    var nombre = document.evaluacion.nom.value;
                    var materia = document.evaluacion.materia.value;
                    var nota1 = parseFloat(document.evaluacion.nota1.value);
                    var nota2 = parseFloat(document.evaluacion.nota2.value);
                    var nota3 = parseFloat(document.evaluacion.nota3.value);

                    var promedio = (nota1 + nota2 + nota3) / 3;
                    document.evaluacion.promedio.value = promedio.toFixed(2);

                    if(promedio >= 14){
                        document.evaluacion.estado.value = "Aprobado";
                        document.evaluacion.mensaje.value = "¡Felicidades " + nombre + "! Has aprobado " + materia;
                    }
                    else if(promedio >= 10){
                        document.evaluacion.estado.value = "Supletorio";
                        document.evaluacion.mensaje.value = nombre +", tienes derecho a supletorio en " + materia;
                    }
                    else{
                        document.evaluacion.estado.value = "Reprobado";
                        document.evaluacion.mensaje.value = "Lo sentimos " + nombre + ", no alcanzaste la nota mínima en " + materia;
                    }
                }
            </script>
